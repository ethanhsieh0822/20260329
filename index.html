<script>

  const btn = document.getElementById('drawBtn');

  const modal = document.getElementById('confirmModal');

  const finalGo = document.getElementById('finalGo');

  const cancelGo = document.getElementById('cancelGo');

  const loader = document.getElementById('loader');

  const display = document.getElementById('displayContent');

  const slogan = document.getElementById('sloganText');

  const legal = document.getElementById('legalNotice');


  // 本地儲存金鑰（更新版本以利重新測試）

  const STORAGE_KEY = 'kc_final_v5';


  const pastResult = localStorage.getItem(STORAGE_KEY);

  if (pastResult) {

    lockUI(pastResult);

  }


  btn.addEventListener('click', () => {

    modal.style.display = 'flex';

  });


  cancelGo.addEventListener('click', () => {

    modal.style.display = 'none';

  });


  finalGo.addEventListener('click', () => {

    modal.style.display = 'none';

    startLottery();

  });


  function startLottery() {

    btn.disabled = true;

    btn.textContent = "PROCESSING...";

    slogan.style.display = "none";

    legal.style.display = "none";

    display.innerHTML = "";

    loader.style.display = "flex";


    // 20% 中獎機率

    const isWin = Math.random() < 0.2; 

    const result = isWin ? "WIN" : "LOSE";


    setTimeout(() => {

      loader.style.display = "none";

      localStorage.setItem(STORAGE_KEY, result);

      renderFinal(result);

      btn.textContent = "FINISHED";

    }, 1500);

  }


  function lockUI(res) {

    slogan.style.display = "none";

    legal.style.display = "none";

    renderFinal(res);

    btn.disabled = true;

    btn.textContent = "FINISHED";

  }


  function renderFinal(res) {

    if (res === "WIN") {

      display.innerHTML = `

        <div class="win-title">中獎 / WIN</div>

        <div class="prize-photo" style="background-image: url('https://i.postimg.cc/rw54Cg8S/suan-rong-zheng-xia.jpg')"></div>

        <div class="prize-name">雙倍份量的蒜蓉蒸蝦</div>

      `;

    } else {

      display.innerHTML = `

        <div class="lose-title">未中獎</div>

        <div class="lose-text">感謝您的參與<br>祝您用餐愉快</div>

      `;

    }

  }

</script>
